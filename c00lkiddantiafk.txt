-- === BIẾN ===
local isEnabled = false
local minimized = false
local secondsElapsed = 0
local clickLock = false
local jumpThread = nil

-- === HÀM XỬ LÝ NHẢY (TỐI ƯU) ===
local function DoJump()
    pcall(function()
        local char = LocalPlayer.Character
        if char then
            local hum = char:FindFirstChild("Humanoid")
            if hum and hum.FloorMaterial ~= Enum.Material.Air then
                hum:ChangeState(Enum.HumanoidStateType.Jumping)
            end
        end
    end)
end

-- === DỪNG HỆ THỐNG ===
local function StopSystem()
    isEnabled = false
    if jumpThread then
        task.cancel(jumpThread)
        jumpThread = nil
    end
    ToggleBtn.BackgroundColor3 = Color3.fromRGB(0, 180, 80)
    ToggleBtn.Text = "▶ KÍCH HOẠT"
    LED.BackgroundColor3 = Color3.fromRGB(255, 50, 50)
    StatusText.Text = "● TRẠNG THÁI: ĐÃ DỪNG"
    StatusText.TextColor3 = Color3.fromRGB(255, 100, 100)
    print("[c00lkidd] Đã dừng AFK.")
end

-- === KHỞI ĐỘNG ===
local function StartSystem()
    if isEnabled then
        StopSystem()
        task.wait(0.1)
    end
    
    isEnabled = true
    ToggleBtn.BackgroundColor3 = Color3.fromRGB(200, 30, 30)
    ToggleBtn.Text = "■ DỪNG"
    LED.BackgroundColor3 = Color3.fromRGB(0, 255, 100)
    StatusText.Text = "● TRẠNG THÁI: ĐANG CHẠY"
    StatusText.TextColor3 = Color3.fromRGB(0, 255, 150)
    secondsElapsed = 0
    
    print("[c00lkidd] Khởi động AFK.")
    
    -- Luồng chính
    jumpThread = task.spawn(function()
        while isEnabled do
            task.wait(1)
            secondsElapsed = secondsElapsed + 1
            
            local hrs = math.floor(secondsElapsed / 3600)
            local mins = math.floor((secondsElapsed % 3600) / 60)
            local secs = secondsElapsed % 60
            TimerLabel.Text = string.format("⏱ %02d:%02d:%02d", hrs, mins, secs)
            
            DoJump()
        end
    end)
end

-- === SỰ KIỆN NÚT ===
ToggleBtn.MouseButton1Click:Connect(function()
    if clickLock then return end
    clickLock = true
    task.spawn(function() task.wait(0.3); clickLock = false end)
    
    if isEnabled then
        StopSystem()
    else
        StartSystem()
    end
end)

-- === NÚT ĐÓNG ===
CloseBtn.MouseButton1Click:Connect(function()
    StopSystem()
    ScreenGui:Destroy()
end)

-- === NÚT THU NHỎ ===
MinBtn.MouseButton1Click:Connect(function()
    minimized = not minimized
    if minimized then
        MainFrame.Size = UDim2.new(0, 320, 0, 45)
        Content.Visible = false
        Footer.Visible = false
        MinBtn.Text = "+"
    else
        MainFrame.Size = UDim2.new(0, 320, 0, 220)
        Content.Visible = true
        Footer.Visible = true
        MinBtn.Text = "−"
    end
end)

-- === RESET KHI RESPAWN ===
LocalPlayer.CharacterAdded:Connect(function()
    task.wait(0.5)
    if isEnabled then
        StopSystem()
        task.wait(0.2)
        StartSystem()
    end
end)
