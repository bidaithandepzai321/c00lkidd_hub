local isEnabled = true
local minimized = false
local clickLock = false
local hue = 0

-- === HÀM CẬP NHẬT UI ===
local function UpdateUI()
    if isEnabled then
        ToggleBtn.BackgroundColor3 = Color3.fromRGB(0, 200, 80)
        ToggleBtn.Text = "▶ FLING: ON"
        LED.BackgroundColor3 = Color3.fromRGB(0, 255, 100)
        LEDGlow.BackgroundColor3 = Color3.fromRGB(0, 255, 100)
        StatusText.Text = "● TRẠNG THÁI: ĐANG BẬT"
        StatusText.TextColor3 = Color3.fromRGB(0, 255, 150)
    else
        ToggleBtn.BackgroundColor3 = Color3.fromRGB(200, 30, 30)
        ToggleBtn.Text = "■ FLING: OFF"
        LED.BackgroundColor3 = Color3.fromRGB(255, 50, 50)
        LEDGlow.BackgroundColor3 = Color3.fromRGB(255, 50, 50)
        StatusText.Text = "● TRẠNG THÁI: ĐÃ TẮT"
        StatusText.TextColor3 = Color3.fromRGB(255, 100, 100)
    end
end

-- === HIỆU ỨNG CẦU VỒNG CHO HEADER VÀ CHỮ ===
RunService.Heartbeat:Connect(function()
    hue = (hue + 0.003) % 1
    local color = Color3.fromHSV(hue, 1, 1)
    
    -- Đổi màu nền Header
    Header.BackgroundColor3 = color
    
    -- Đổi màu chữ Title
    Title.TextColor3 = color
    
    -- Đổi màu chữ Footer
    local footerHue = (hue + 0.3) % 1
    Footer.TextColor3 = Color3.fromHSV(footerHue, 0.8, 1)
    
    -- Đổi màu chữ SpeedLabel
    local speedHue = (hue + 0.5) % 1
    SpeedLabel.TextColor3 = Color3.fromHSV(speedHue, 0.9, 1)
end)

-- === XỬ LÝ FLING ===
Workspace.ChildAdded:Connect(function(child)
    if child.Name == "GrabParts" then
        local grabPart = child:FindFirstChild("GrabPart")
        if not grabPart then return end
        
        local weld = grabPart:FindFirstChild("WeldConstraint")
        if not weld or not weld.Part1 then return end
        
        local targetPart = weld.Part1
        local bodyVelocity = Instance.new("BodyVelocity")
        bodyVelocity.Parent = targetPart
        
        child:GetPropertyChangedSignal("Parent"):Connect(function()
            if not child.Parent then
                if isEnabled then
                    local speed = tonumber(SpeedBox.Text) or 850
                    if speed < 100 then speed = 100 end
                    if speed > 5000 then speed = 5000 end
                    
                    bodyVelocity.MaxForce = Vector3.new(1e9, 1e9, 1e9)
                    bodyVelocity.Velocity = Workspace.CurrentCamera.CFrame.LookVector * speed
                else
                    bodyVelocity.MaxForce = Vector3.zero
                end
                Debris:AddItem(bodyVelocity, 1)
            end
        end)
    end
end)

-- === SỰ KIỆN NÚT ===
ToggleBtn.MouseButton1Click:Connect(function()
    if clickLock then return end
    clickLock = true
    task.spawn(function() task.wait(0.3); clickLock = false end)
    
    isEnabled = not isEnabled
    UpdateUI()
end)

-- === NÚT ĐÓNG ===
CloseBtn.MouseButton1Click:Connect(function()
    ScreenGui:Destroy()
end)

-- === NÚT THU NHỎ ===
MinBtn.MouseButton1Click:Connect(function()
    minimized = not minimized
    if minimized then
        MainFrame.Size = UDim2.new(0, 360, 0, 50)
        Content.Visible = false
        Footer.Visible = false
        MinBtn.Text = "+"
    else
        MainFrame.Size = UDim2.new(0, 360, 0, 260)
        Content.Visible = true
        Footer.Visible = true
        MinBtn.Text = "−"
    end
end)

-- === CẬP NHẬT TỐC ĐỘ ===
SpeedBox.FocusLost:Connect(function()
    local val = tonumber(SpeedBox.Text)
    if not val or val < 100 then
        SpeedBox.Text = "100"
    elseif val > 5000 then
        SpeedBox.Text = "5000"
    end
    SpeedLabel.Text = "⚡ TỐC ĐỘ: " .. SpeedBox.Text
end)

-- === HIỆU ỨNG HOVER ===
ToggleBtn.MouseEnter:Connect(function()
    if isEnabled then
        ToggleBtn.BackgroundColor3 = Color3.fromRGB(0, 220, 100)
    else
        ToggleBtn.BackgroundColor3 = Color3.fromRGB(220, 40, 40)
    end
end)

ToggleBtn.MouseLeave:Connect(function()
    UpdateUI()
end)

-- === KHỞI TẠO UI ===
UpdateUI()
